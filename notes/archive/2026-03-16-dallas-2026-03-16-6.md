```bash
TOKEN=$(az account get-access-token --resource 499b84ac-1321-427f-aa17-267ca6975798 --query accessToken -o tsv)
cat > /tmp/pr-thread.json << 'JSONEOF'
{ "comments": [{ "parentCommentId": 0, "content": "...", "commentType": 1 }], "status": 1 }
JSONEOF
curl -s -X POST "https://office.visualstudio.com/DefaultCollection/{project}/_apis/git/repositories/{repoId}/pullRequests/{prId}/threads?api-version=7.1" \
  -H "Authorization: Bearer $TOKEN" -H "Content-Type: application/json" -d @/tmp/pr-thread.json
```
